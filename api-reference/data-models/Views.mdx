## `Views` Class

```jsx
export declare class Views {
    comment?: {
        [commentAnnotationId: string]: {
            views: {
                [userSnippylyId: string]: View;
            };
            comments: {
                [commentId: string]: {
                    views: {
                        [userSnippylyId: string]: View;
                    };
                };
            };
        };
    };
}
export declare class View {
    timestamp: any;
}
```
