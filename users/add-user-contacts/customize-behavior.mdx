## Reset Contact List

To reset a user's contact list, use the `client.identify()` mehtod and pass in `{replaceContacts: true}` as the second parameter.

```jsx
client.identify(yourLoggedInUser, {
    replaceContacts: true
}).then((res) => {
  // User auth successful with Velt
}).catch((err) => {
  // Failed to authenticate user with Velt
});
```

## Reset Contact List using POST Method API

You can also replace an entire Group Ids contact list using a POST Method on the following API endpoint.

Endpoint:
```
https://updatecontactlist-4mfhcuyw2q-uc.a.run.app
```

Header:
```jsx
Authorization: Bearer {AUTH_TOKEN}
```

Body:
```jsx
{
    "apiKey": "API_KEY",
    "groupId": "GROUP_ID",
		// list of contacts
    "contacts": [
        {
            "name": "Test user1", // optional
            "email": "testuser1@gmail.com", // required
						"userId": "userId", // optional
```

```jsx React / Next.js
import { useVeltClient } from '@veltdev/react';
import { useEffect} from '/react';


function YourComponent() {

    const { client } = useVeltClient();

    useEffect(() => {
        if(client){
            //Reset Contact List
            client.identify(yourLoggedInUser, {
                replaceContacts: true
            }).then((res) => {
            // User auth successful with Velt
            }).catch((err) => {
            // Failed to authenticate user with Velt
            });
        }
    },[client]);

  return (
    <div>

    </div>
  )
}
```
