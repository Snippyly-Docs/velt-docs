<Tabs>
    <Tab title="React / Next.js">

<Steps titleSize="h2">
<Step title="Enable Audio Only Mode">
Sets the mode to audio only mode.

This is the default mode if no type is provided.

```jsx
<VeltHuddleTool type="audio"/>
```
</Step>

<Step title="Enable Video Only Mode">
Sets the mode to video only mode.

```jsx
<VeltHuddleTool type="video"/>
```
</Step>

<Step title="Enable Screen Only Mode">
Sets the mode to screen only mode.

```jsx
<VeltHuddleTool type="screen"/>
```
</Step>

<Step title="Enable All Modes">
Sets the mode to all modes.

```jsx
<VeltHuddleTool type="all"/>
```
</Step>


<Step title="Dark Mode">
Whether dark mode is enabled.

`Default: false`

```js
<VeltHuddle darkMode={true} />
````

</Step>

</Steps>




</Tab>


<Tab title="HTML">
<Steps titleSize="h2">
<Step title="Enable Audio Only Mode">
Sets the mode to audio only mode.

This is the default mode if no type is provided.

```jsx
<velt-huddle-tool type="audio"></velt-huddle-tool>
```
</Step>

<Step title="Enable Video Only Mode">
Sets the mode to video only mode.

```jsx
<velt-huddle-tool  type="video"></velt-huddle-tool>
```
</Step>

<Step title="Enable Screen Only Mode">
Sets the mode to screen only mode.

```jsx
<velt-huddle-tool type="screen"></velt-huddle-tool>
```
</Step>

<Step title="Enable All Modes">
Sets the mode to all modes.

```jsx
<velt-huddle-tool  type="all"></velt-huddle-tool>
```
</Step>


<Step title="Dark Mode">
Whether dark mode is enabled.

`Default: false`

```js
<velt-huddle dark-mode="true"></velt-huddle>
````

</Step>

</Steps>


</Tab>

</Tabs>




<RequestExample>

```js React / Next.js
import { VeltHuddle, VeltHuddleTool} from '@veltdev/react';

export default function App() {

  return (
    <VeltHuddle darkMode={true}/> 

    <div className="toolbar">
      <VeltHuddleTool type='audio' /> 
      <VeltHuddleTool type='video' /> 
      <VeltHuddleTool type='screen' /> 
      <VeltHuddleTool type='all' /> 
    </div>
  );
}
```


```html HTML
<div>
  <velt-huddle dark-mode='true'></velt-huddle>
  <div class="toolbar">
    <velt-huddle-tool type='audio'></velt-huddle-tool> 
    <velt-huddle-tool type='video'></velt-huddle-tool> 
    <velt-huddle-tool type='screen'></velt-huddle-tool> 
    <velt-huddle-tool type='all'></velt-huddle-tool> 
  </div>
</div>

```

</RequestExample>
